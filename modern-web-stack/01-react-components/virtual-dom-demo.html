// Traditional: Direct DOM manipulation let traditionalItemCount = 0; function addTraditionalItem() { traditionalItemCount++; document.getElementById('traditionalItemCount').textContent = traditionalItemCount; const list = document.getElementById('traditionalList'); const item = document.createElement('div'); item.className = 'list-item traditional'; item.textContent = `Item ${traditionalItemCount}`; list.appendChild(item); // This causes immediate reflow and repaint } // React: Virtual DOM function ItemList() { const [items, setItems] = React.useState([]); const addItem = () => { setItems([...items, `Item ${items.length + 1}`]); // React batches updates and optimizes rendering }; return ( <div> <p>Items: <span>{items.length}</span></p> <button className="bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-600" onClick={addItem} > Add Item </button> <div className="list-demo"> {items.map((item, index) => ( <div key={index} className="list-item modern"> {item} </div> ))} </div> </div> ); }
